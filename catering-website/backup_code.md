}

function updateGuests() {
    guestCount = document.getElementById('guest-slider').value;
    document.getElementById('guest-count').innerText = guestCount;
    document.getElementById('sheet-total-guests').innerText = guestCount;
    calculateTotal();
}

function toggleItem(id, price, name) {
    const idx = selectedItems.findIndex(x => x.id === id);
    if(idx > -1) selectedItems.splice(idx, 1);
    else selectedItems.push({ id, price, name });
    
    calculateTotal();
    drawVisualPlate();
}

function calculateTotal() {
    const perPlate = selectedItems.reduce((sum, item) => sum + item.price, 0);
    const grandTotal = perPlate * guestCount;
    
    document.getElementById('sheet-per-plate').innerText = 'â‚¹' + perPlate.toLocaleString();
    document.getElementById('sheet-grand-total').innerText = 'â‚¹' + grandTotal.toLocaleString();
}

function drawVisualPlate() {
    const svgArea = document.getElementById('visual-plate-svg');
    const legend = document.getElementById('visual-plate-legend');
    
    if(selectedItems.length === 0) {
        svgArea.innerHTML = '';
        legend.innerHTML = '<p>Select items to visualize plate.</p>';
        return;
    }

    let svg = `<svg viewBox="0 0 100 100" width="100%" height="100%">
        <circle cx="50" cy="50" r="48" fill="#fff" stroke="#D4AF37" stroke-width="1.5"/>
        <circle cx="50" cy="50" r="33" fill="none" stroke="#D4AF37" stroke-width="0.5" stroke-dasharray="2,2"/>
    `;
    legend.innerHTML = '<ul>';
    const colors = ['#C5A059', '#1A1A1A', '#555555', '#D4AF37']; // Taj Theme Colors
    
    selectedItems.forEach((m, i) => {
        const angle = (i / selectedItems.length) * Math.PI * 2;
        const cx = 50 + 20 * Math.cos(angle);
        const cy = 50 + 20 * Math.sin(angle);
        const color = colors[i % colors.length];
        
        svg += `<circle cx="${cx}" cy="${cy}" r="4" fill="${color}" />`;
        legend.innerHTML += `<li><span class="legend-dot" style="background:${color}"></span>${m.name}</li>`;
    });
    
    svg += `</svg>`;
    svgArea.innerHTML = svg;
    legend.innerHTML += '</ul>';
}

function secureBooking() {
    const date = document.getElementById('event-date').value;
    const location = document.getElementById('event-location').value;
    
    if(!date || selectedItems.length === 0) {
        alert("Please set Event Date and select at least 1 menu item.");
        return;
    }
    
    const perPlate = selectedItems.reduce((sum, item) => sum + item.price, 0);
    const grandTotal = perPlate * guestCount;
    
    db_leads.unshift({ 
        id: 'L'+Date.now(), client: 'Walk-in Royal Client', date, guests: guestCount, total: grandTotal, status: 'Pending Approval' 
    });
    localStorage.setItem('v4_leads', JSON.stringify(db_leads));
    
    alert("Your Royal Booking Request has been secured! Our Executive will call you shortly.");
    window.location.reload();
}

function updateLeadStatus(id, newStatus) {
    const idx = db_leads.findIndex(x => x.id === id);
    db_leads[idx].status = newStatus;
    localStorage.setItem('v4_leads', JSON.stringify(db_leads));
    renderLedger(); // Refresh background ledger
}
